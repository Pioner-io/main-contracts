const { expect } = require("chai");
const { ethers } = require("hardhat");

describe("fakeUSD Deployment", function () {
    it("Should deploy and have correct name and symbol", async function () {
      const FakeUSD = await ethers.getContractFactory("fakeUSD");
      const fakeUSD = await FakeUSD.deploy();
  
      expect(await fakeUSD.name()).to.equal("Fake USD");
      expect(await fakeUSD.symbol()).to.equal("fUSD");
    });
  });
  

describe("Minting", function () {
  it("Should mint the specified amount of fUSD to an address", async function () {
    const [owner, addr1, addr2] = await ethers.getSigners();
    const FakeUSD = await ethers.getContractFactory("fakeUSD");
    const fakeUSD = await FakeUSD.deploy();
    
    const mintAmount = ethers.parseUnits("100", 18);
    await fakeUSD.mint(addr1.address, mintAmount);

    expect(await fakeUSD.balanceOf(addr1.address)).to.equal(mintAmount);
  });
});

  
describe("Unauthorized Minting", function () {
    it("Should fail if non-owner tries to mint", async function () {
      const [owner, addr1] = await ethers.getSigners();
      const FakeUSD = await ethers.getContractFactory("fakeUSD");
      const fakeUSD = await FakeUSD.deploy();
  
      const mintAmount = ethers.parseUnits("100", 18);
  
      await expect(fakeUSD.connect(addr1).mint(addr1.address, mintAmount))
        .to.be.revertedWith('Some error message related to permission');
    });
  });
